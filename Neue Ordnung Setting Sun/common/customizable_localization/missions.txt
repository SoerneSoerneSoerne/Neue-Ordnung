country_mission_id = { #corresponds to the journal entry's mission id, if equal mission will be shown
	type = country
	random_valid = no

	text = {
		trigger = {
			OR = {
			   AND = {
			    exists = c:SPG
				this = c:SPG
				exists = c:GNZ
				this = c:GNZ
				exists = c:GLF
				this = c:GLF
				exists = c:BGF
				this = c:BGF
				exists = c:WEH
				this = c:WEH
				exists = c:SSG
				this = c:SSG
				exists = c:GFF
				this = c:GFF
				exists = c:GGV
				this = c:GGV
				exists = c:RKO
				this = c:RKO
				exists = c:RKU
				this = c:RKU
				exists = c:RKM
				this = c:RKM
				exists = c:RKK
				this = c:RKK
				exists = c:NZI
				this = c:NZI
				}
			}
		}
		localization_key = mission_key_1
	}

	text = {
		localization_key = mission_key_no_missions
	}
}
